	}
	option = {
		ai_chance = {
			base = 75
		}		
		name = "Refuse Request"	
		FROM = { 
			add_opinion_modifier = { target = ROOT modifier = refused_lease_request }
			phone_notification_generic_request_refused = yes
		}
		clear_variable = requested_state_lease
	}
}
