country_event = {
	id = ccw_events.10		#China partition resolution implementation
	title = ccw_events.10.t
	desc = ccw_events.10.d
	picture = GFX_report_event_generic_sign_treaty2

	is_triggered_only = yes

	option = {
		name = "There is only one China."
		ai_chance = {
			modifier = {
				factor = 25
				ROOT = { surrender_progress < 0.1 }
			}
		}
		unsc_mark_for_sanctions = yes
		hidden_effect = {
			add_named_threat = { threat = 5 name = UNSC_REJECTED_PARTITION_PLAN }
			news_event = { id = ccw_events.12 days = 5 }		
			cw_unsc_clear_junk = yes
			clr_country_flag = unsc_partition_resolution_china
			set_country_flag = rejected_china_partition
		}
	}
	option = {
		name = "Agree to partition plan."
		ai_chance = {
			modifier = {
				factor = 25
				ROOT = { surrender_progress > 0.3 }
			}
			modifier = {
				add = -50
				is_historical_focus_on = yes
			}
		}
		set_country_flag = agreed_china_partition
		if = {
			limit = {
				AND = {
					PRC = { has_country_flag = agreed_china_partition }
					CHI = { has_country_flag = agreed_china_partition }
				}
			}
			hidden_effect = {
				every_state = { limit = { is_controlled_by = CHI } CHI = { set_state_owner = PREV } }
				every_state = { limit = { is_controlled_by = PRC } PRC = { set_state_owner = PREV } }
				CHI = { white_peace = PRC }
				clr_country_flag = unsc_partition_resolution_china
			}
			news_event = { id = ccw_events.11 days = 5 }
		}
		if = {
			limit = {
				OR = {
					PRC = { has_country_flag = rejected_china_partition }
					CHI = { has_country_flag = rejected_china_partition }
				}			
			}
			hidden_effect = {
				cw_unsc_clear_junk = yes
				clr_country_flag = unsc_partition_resolution_china
				clr_global_flag = unsc_implementing_resolution_flag
			}
		}
	}	
}
