country_event = {
	id = prc_events.3		#Great Leap Forward Famine
	title = "A Famine Grips the Country"
	desc = prc_events.3.d
	picture = GFX_harvest_image

	is_triggered_only = yes

	option = {
		name = "The Great Leap Must Continue!"
		random_list = {
			90 = {
				cw_apply_severe_famine = yes
				add_manpower = -3000000
				subtract_from_variable = { great_leap_forward_population = 0.35 }
				add_popularity = {
					ideology = communism
					popularity = 0.10
				}
				add_to_variable = { great_leap_forward_reformist_popularity = 0.35 }
			}
			10 = {
				cw_apply_mild_famine = yes
				add_manpower = -1000000
				subtract_from_variable = { great_leap_forward_population = 0.15 }
				add_popularity = {
					ideology = communism
					popularity = 0.05
				}				
				add_to_variable = { great_leap_forward_reformist_popularity = 0.15 }
			}
		}
		every_owned_state = {
			limit = {
				has_state_flag = killed_sparrows
			}
			add_resource = {
				type = crops
				amount = -5
			}
			clr_state_flag = killed_sparrows
		}
	}
}
