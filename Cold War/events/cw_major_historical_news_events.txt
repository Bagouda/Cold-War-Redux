				set_variable = { country_ref_second_option_votes_percent = 30 }	#Beria

				add_ideas = sov_dm_bulganin
				add_ideas = sov_hog_georgi_malenkov
				add_ideas = sov_tm_alexei_kosygin
				add_ideas = sov_fm_vyacheslav_molotov
				if = {
					limit = {
						has_completed_focus = SOV_the_old_guard
					}
					add_ideas = sov_power_struggle_reform_opposition
					remove_ideas = political_power_struggle
					remove_ideas = sov_leningrad_plot
					set_party_name = { ideology = communism long_name = "The Anti Party Group" name = "Anti Party Group" }
