country_event = {		# Government Instability
	id = cw_war_exaustion_events.5
	title = cw_war_exaustion_events.5.t
	desc = cw_war_exaustion_events.5.d
	picture = GFX_event_peaceful_protest

	is_triggered_only = yes

	option = {
		name = "Ok"
		if = {
			limit = {
				OR = {
					has_government = democratic
					has_government = democratic_socialist
				}
			}
			add_timed_idea = { idea = unstable_government_communist days = 180 }
			else  = {
				add_timed_idea = { idea = unstable_government days = 180 }				
			}
		}
		add_popularity = { ideology = var:highest_party_ideology@exclude_ruling_party popularity = 0.15 }
		random_list = {
			50 = {
				cw_remove_defense_advisor = yes
			}
			50 = {
				cw_remove_army_chief_advisor = yes
			}
		}		
	}
}
