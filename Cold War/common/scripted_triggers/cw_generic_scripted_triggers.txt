same_ideology_check = {
	OR = {
		AND = {
			ROOT = { has_government = neutrality }
			THIS = { has_government = neutrality }
		}
		AND = {
			ROOT = { has_government = unaligned_right }
			THIS = { has_government = unaligned_right }
		}		
		AND = {
			ROOT = { has_government = communism_totalitarian }
			THIS = { has_government = communism_totalitarian }
		}			
		AND = {
			ROOT = { has_government = communism }
			THIS = { has_government = communism }
		}	
		AND = {
			ROOT = { OR = { has_government = democratic has_government = democratic_socialist } }
			THIS = { OR = { has_government = democratic has_government = democratic_socialist } }
		}
	}
}
similar_ideology_check = {
	custom_trigger_tooltip = {
		tooltip = HAS_SAME_IDEOLOGY_GROUP
		OR = {
			AND = {
				ROOT = { has_government = communism_totalitarian }
				THIS = { OR = { has_government = communism has_government = communism_totalitarian has_government = neutrality } }
			}
			AND = {
				ROOT = { has_government = communism }
				THIS = { OR = { has_government = communism has_government = communism_totalitarian has_government = neutrality } }
			}
			AND = {
				ROOT = { has_government = neutrality }
				THIS = { OR = { has_government = communism has_government = communism_totalitarian has_government = neutrality } }
			}
			AND = {
				ROOT = { OR = { has_government = democratic has_government = democratic_socialist has_government = unaligned_right } }
				THIS = { OR = { has_government = colonial has_government = democratic has_government = democratic_socialist has_government = unaligned_right } }
			}
			AND = {
				THIS = { has_government = unaligned_right }
				THIS = { OR = { has_government = democratic has_government = democratic_socialist has_government = unaligned_right } }
			}
		}
	}
}

same_ideology_org_checker = {
	OR = {
		AND = {
			ROOT = { has_government = communism_totalitarian }
			THIS = { communism_totalitarian < 0.30 }
		}			
		AND = {
			ROOT = { has_government = communism }
			THIS = { communism < 0.30 }
		}
		AND = {
			ROOT = { has_government = neutrality }
			THIS = { OR = { neutrality < 0.30  unaligned_right < 0.30 } }
		}		
		AND = {
			ROOT = { OR = { has_government = democratic has_government = democratic_socialist } }
			THIS = { OR = { communism_totalitarian > 0.10 communism > 0.10 } }
		}		
	}
}
