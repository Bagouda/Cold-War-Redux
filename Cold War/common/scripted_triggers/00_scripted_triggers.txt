is_democracy = {
	custom_trigger_tooltip = {
		tooltip = "Is a democratic regime"
		NOT = {
			OR = {
				has_government = communism
				has_government = communism_totalitarian
				has_government = fascism
				has_government = unaligned_right
			}
		
		}
	}
	
}

		}
	}

}
