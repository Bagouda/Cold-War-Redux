	cw_famine_import_crops_from_abroad = {
		
		icon = GFX_bribe_decision_icon
		allowed = {}

		available = {
		}

		visible = {
			OR = {
				has_idea = cw_famine_mild
				has_idea = cw_famine_severe
			}
		}
		
		target_trigger = {
			FROM = {
				is_major = yes
				has_opinion = { target = ROOT value > 0 }
				NOT = {
					has_war_with = ROOT
					is_in_array = {
						array = ROOT.embargoed_countries
						value = THIS.id
					}
					ROOT = {
						is_in_array = {
							array = FROM.embargoed_countries
							value = THIS.id
						}
					}
					OR = {
						has_idea = cw_famine_mild
						has_idea = cw_famine_severe
					}
				}
				check_variable = { resource_produced@crops > 300 }
			}
		}
		
		cancel_trigger = {
			custom_trigger_tooltip = {
				tooltip = "[THIS.GetName] is embargoed by [FROM.GetName]"
				ROOT = {
					is_in_array = {
						array = FROM.embargoed_countries
						value = THIS.id
					}
				}
			}
		}
		
		modifier = {
			drift_defence_factor = -0.10
			country_resource_crops = 75
			country_resource_cost_crops = -100
		}
		
		cost = 200
		days_re_enable = 0
		days_remove = 180

		complete_effect = {
			custom_effect_tooltip = CW_SUBSIDISE_FARMERS_EFFECT_TT
			if = {
				limit = {
					has_active_mission = cw_generic_mild_famine_timer
				}
				add_days_mission_timeout = { mission = cw_generic_mild_famine_timer days = 100 }
			}
			if = {
				limit = {
					has_active_mission = cw_generic_severe_famine_timer
				}
				add_days_mission_timeout = { mission = cw_generic_severe_famine_timer days = 100 }
			}
			every_owned_state = { limit = { has_resources_amount = { resource = crops amount > 0 } } add_state_modifier = { modifier = { local_resources = 0.1 } } }
			hidden_effect = {
				### NOTIFICATION ###
				FROM = {
					set_country_flag = { flag = phone_notification_food_exports value = 1 days = 5 }
					country_phone_alert = yes
				}
				### NOTIFICATION ###
				add_to_variable = { crops_import_consumer_goods@FROM = 0.05 }
				if = {
					limit = {
						FROM = { has_government = communism }
					}
					add_to_variable = { crops_import_communist_ref_influence@FROM = 0.1 }
				}
				if = {
					limit = {
						FROM = { has_government = communism_totalitarian }
					}
					add_to_variable = { crops_import_communist_total_influence@FROM = 0.1 }
				}
				if = {
					limit = {
						FROM = { has_government = democratic }
					}
					add_to_variable = { crops_import_dem_right_influence@FROM = 0.1 }
				}
				if = {
					limit = {
						FROM = { has_government = democratic_socialist }
					}
					add_to_variable = { crops_import_dem_left_influence@FROM = 0.1 }
				}
				if = {
					limit = {
						FROM = { has_government = unaligned_right }
					}
					add_to_variable = { crops_import_unaligned_right_influence@FROM = 0.1 }
				}
				if = {
					limit = {
						FROM = { has_government = neutrality }
					}
					add_to_variable = { crops_import_unaligned_left_influence@FROM = 0.1 }
				}
				ROOT = { add_dynamic_modifier = { modifier = cw_importing_food_from_abroad scope = FROM } }
				set_temp_variable = { food_import_cost = crops_import_consumer_goods@FROM }
				multiply_temp_variable = { food_import_cost = -1 }
				FROM = {
					add_to_variable = { food_export_cost@ROOT = food_import_cost }
					add_dynamic_modifier = { modifier = cw_exporting_food scope = ROOT } 
				}
			}
		}
		remove_effect = {
			hidden_effect = {
				custom_effect_tooltip = CW_SUBSIDISE_FARMERS_REMOVE_TT
				every_owned_state = { limit = { has_resources_amount = { resource = crops amount > 0 } } add_state_modifier = { modifier = { local_resources = -0.1 } } }
				remove_dynamic_modifier = {
					modifier = cw_importing_food_from_abroad
					scope = FROM
				}
				FROM = { remove_dynamic_modifier = { modifier = cw_exporting_food scope = ROOT } }
			}
		}
		ai_will_do = {
			factor = 1
		}
